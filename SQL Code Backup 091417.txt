SQL - Inventory:

SELECT GLOVIA_PROD_VW_INVENTORY_COMB.ITEM, GLOVIA_PROD_ITEM.DESCRIPTION AS ITEM_DESC, GLOVIA_PROD_VW_INVENTORY_COMB.COMBINE_QTY AS ON_HAND_QTY, GLOVIA_PROD_VW_INVENTORY_COMB.STATUS, GLOVIA_PROD_VW_INVENTORY_COMB.LIST_PRICE AS PRICE, "" AS [IMAGE]
FROM GLOVIA_PROD_VW_INVENTORY_COMB INNER JOIN GLOVIA_PROD_ITEM ON GLOVIA_PROD_VW_INVENTORY_COMB.ITEM = GLOVIA_PROD_ITEM.ITEM
WHERE (((GLOVIA_PROD_VW_INVENTORY_COMB.CCN)="MAXIM"))
GROUP BY GLOVIA_PROD_VW_INVENTORY_COMB.ITEM, GLOVIA_PROD_ITEM.DESCRIPTION, GLOVIA_PROD_VW_INVENTORY_COMB.COMBINE_QTY, GLOVIA_PROD_VW_INVENTORY_COMB.STATUS, GLOVIA_PROD_VW_INVENTORY_COMB.LIST_PRICE, ""
HAVING (((GLOVIA_PROD_VW_INVENTORY_COMB.ITEM) Not Like "E*"))
ORDER BY GLOVIA_PROD_VW_INVENTORY_COMB.LIST_PRICE DESC;

SQL - Sales Orders: 

SELECT GLOVIA_PROD_C_TBL_COMMREPORT.CREATED_DATE AS [DATE], GLOVIA_PROD_C_TBL_COMMREPORT.BRANCH AS SALES_REP, GLOVIA_PROD_C_TBL_COMMREPORT.CUSTOMER,
GLOVIA_PROD_C_TBL_COMMREPORT.CUS_BUY_NAME AS CUSTOMER_NAME, Trim([GLOVIA_PROD_C_TBL_COMMREPORT].[SO]) AS SO_NUM, GLOVIA_PROD_C_TBL_COMMREPORT.SO_LINE AS SO_LN,
Trim([AR_DOC]) AS INV_NUM, GLOVIA_PROD_C_TBL_COMMREPORT.AR_DOC_LINE AS INV_LN, GLOVIA_PROD_C_TBL_COMMREPORT.ITEM, GLOVIA_PROD_C_TBL_COMMREPORT.QTY AS QTY_PRC,
GLOVIA_PROD_C_TBL_COMMREPORT.UNIT_PRICE AS PRICE FROM GLOVIA_PROD_C_TBL_COMMREPORT WHERE (((GLOVIA_PROD_C_TBL_COMMREPORT.CCN)="MAXIM") AND
((GLOVIA_PROD_C_TBL_COMMREPORT.AR_DOC_TYPE)="I")) GROUP BY GLOVIA_PROD_C_TBL_COMMREPORT.CREATED_DATE, GLOVIA_PROD_C_TBL_COMMREPORT.BRANCH,
GLOVIA_PROD_C_TBL_COMMREPORT.CUSTOMER, GLOVIA_PROD_C_TBL_COMMREPORT.CUS_BUY_NAME, Trim([GLOVIA_PROD_C_TBL_COMMREPORT].[SO]), GLOVIA_PROD_C_TBL_COMMREPORT.SO_LINE,
Trim([AR_DOC]), GLOVIA_PROD_C_TBL_COMMREPORT.AR_DOC_LINE, GLOVIA_PROD_C_TBL_COMMREPORT.ITEM, GLOVIA_PROD_C_TBL_COMMREPORT.QTY, GLOVIA_PROD_C_TBL_COMMREPORT.UNIT_PRICE
HAVING (((GLOVIA_PROD_C_TBL_COMMREPORT.CREATED_DATE) Between #3/1/2015# And #5/31/2015#));


SQL - Quotas:
SELECT tblQuota.Branch AS SALES_REP, tblQuota.CurrYrQuota AS YEARLY_QUOTA, tblQuota.CurYrMoQuota AS MONTHLY_QUOTA, GLOVIA_PROD_SLS_REP.NAME AS REPNAME
FROM GLOVIA_PROD_SLS_REP INNER JOIN tblQuota ON GLOVIA_PROD_SLS_REP.SALES_REP = tblQuota.Branch
WHERE (((GLOVIA_PROD_SLS_REP.CCN)="MAXIM"))
ORDER BY tblQuota.Branch;
